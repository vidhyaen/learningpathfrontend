mport { useState } from 'react';

function CodeSnippet({ code }) {
  const [copied, setCopied] = useState(false);

  function handleCopy() {
    navigator.clipboard.writeText(code)
      .then(() => setCopied(true))
      .catch(() => setCopied(false));
  }

  return (
    <div>
      <pre>
        <code>{code}</code>
      </pre>
      <button onClick={handleCopy}>
        {copied ? 'Copied!' : 'Copy Code'}
      </button>
    </div>
  );
  
}
